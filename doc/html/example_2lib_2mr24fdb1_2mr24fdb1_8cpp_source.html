191 MR24FDB1::Result_t MR24FDB1::readPacket()
192 {
193  uint8_t inByte; // incoming byte
194  int lenCount = 0; // lenght counter
195  bool done = false;
196  Result_t retVal=MR24FDB1::Res_Ok;
197  uint8_t *pp;
198  uint16_t crc;
199  uint16_t pktCrc;
200  uint16_t pktLength = 0;
201 
202  unsigned long time;
203  unsigned long maxTime;
204 
205  time = millis();
206  maxTime = 1000; // max millisec to wait
207 
208  PacketReceived = false;
209 
210  // wait for header
211  inByte = 0;
212 
213  DBG3("readPacket: Waiting for start code\n");
214  while (inByte != MESSAGE_HEAD) {
215  if((retVal=readByte(&inByte)) != Res_Ok) return retVal;
216  }
217  Packet.frameStart=inByte;
218 
219  DBG3("Getting Length....:");
220 
221  // got length bytes yet ?
222 
223  while(lenCount<2) {
224  if((retVal=readByte(&inByte)) != Res_Ok) return retVal;
225  pktLength |= inByte<<(8*lenCount);
226  lenCount++;
227  }
228 
229  if (pktLength > MAXPACKET_LENGTH) {
230  DBG3NL;
231  ERROR("Length (%d) to high, Aborting\n",pktLength);
232  return Res_PacketToLong;
233  }
234 
235  DBG3V_S("%d (%4.4x)",pktLength,pktLength);
236  DBG3NL;
237 
238  Packet.length=pktLength;
239 
240  DBG3("Getting rest of packet\n");
241  pp = ((uint8_t *) &Packet.funcClass) ; // step over length and header
242 
243  for (int pi = 2; pi <= pktLength;pi++) { // pi=2 -> we alreade received 2 bytes and header doesn count
244  if((retVal=readByte(&inByte)) != Res_Ok) return retVal;
245  *pp++=inByte;
246  }
247 
248  uint8_t *RawPkt = (uint8_t *)&Packet;
249  pktCrc=RawPkt[pktLength] | (RawPkt[pktLength-1]<<8); // get crc at end of received packet
250  // and yes pktLength is a coorect index since the received length
251  // does not count the header byte...
252  crc=calculateCrc16(); // Calculate actual crc on received packet
253  if (pktCrc != crc){
254  DBG3NL;
255  ERROR("CRC Check Failed: PktCrc: %d (%4.4x), CaclCrc: %d (%4.4x)\n", pktCrc,pktCrc,crc,crc);
256  throw MR24FDB1_Exception( Res_InvCrc,0,0,0,crc);
257  return Res_InvCrc;
258  }
259 
260  PacketReceived = true;
261  return Res_Ok;
262 }
