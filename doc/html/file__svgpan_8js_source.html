159 function doZoom(g,point,zoomFactor)
160 {
161  var p = point.matrixTransform(g.getCTM().inverse());
162  var k = root.createSVGMatrix().translate(p.x, p.y).scale(zoomFactor).translate(-p.x, -p.y);
163  var n = g.getCTM().multiply(k);
164  var s = Math.max(n.a,n.d);
165  if (s>maxZoom) n=n.translate(p.x,p.y).scale(maxZoom/s).translate(-p.x,-p.y);
166  else if (s<minZoom) n=n.translate(p.x,p.y).scale(minZoom/s).translate(-p.x,-p.y);
167  setCTM(g, n);
168  stateTf = stateTf.multiply(n.inverse());
169 }
